/** * ระบบทะเบียนผู้ป่วย Cloud V.Ultimate Professional * Backend Configuration & API Logic */ const SETTINGS_SHEET = 'Settings'; function doPost(e) { try { const postData = JSON.parse(e.postData.contents); const action = postData.action; const data = postData.data; let result; if (action === 'getDropdownData') result = getDropdownData(); else if (action === 'getSheetNames') result = getSheetNames(); else if (action === 'getNextSeq') result = getNextSeq(data); else if (action === 'getAllData') result = getAllData(data); else if (action === 'processData') result = processData(data); else if (action === 'closeMonthAction') result = closeMonthAction(data); else if (action === 'getMonthlyReportData') result = getMonthlyReportData(data); // รายงานรวมยอด else if (action === 'createPatientPDF') result = createPatientPDF(data); else if (action === 'checkDuplicateAN') result = checkDuplicateAN(data); else throw new Error("Action not found"); return ContentService.createTextOutput(JSON.stringify({ status: 'success', data: result })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function getDropdownData() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SETTINGS_SHEET); if (!sheet) return { sai: [], sithi: [], ward: [], dx: [] }; const data = sheet.getRange(2, 1, sheet.getLastRow(), 5).getDisplayValues(); let dropdowns = { sai: [], sithi: [], ward: [], dx: [] }; data.forEach(row => { if(row[0]) dropdowns.sai.push(row[0]); if(row[1]) dropdowns.sithi.push(row[1]); if(row[2]) dropdowns.ward.push(row[2]); if(row[4]) dropdowns.dx.push(row[4]); }); return dropdowns; } function getSheetNames() { const ss = SpreadsheetApp.getActiveSpreadsheet(); return ss.getSheets().map(s => s.getName()).filter(name => name !== SETTINGS_SHEET).reverse(); } function getNextSeq(sheetName) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(sheetName); if (!sheet || sheet.getLastRow() < 2) return 1; const values = sheet.getRange(2, 1, sheet.getLastRow()-1, 1).getValues(); const max = values.reduce((m, r) => { let val = parseInt(r[0]); return !isNaN(val) && val > m ? val : m; }, 0); return max + 1; } function getMonthStartDate(sheetName) { const months = ["มกราคม", "กุมภาพันธ์", "มีนาคม", "เมษายน", "พฤษภาคม", "มิถุนายน", "กรกฎาคม", "สิงหาคม", "กันยายน", "ตุลาคม", "พฤศจิกายน", "ธันวาคม"]; let mIdx = -1; months.forEach((m, i) => { if (sheetName.includes(m)) mIdx = i; }); if (mIdx === -1) return null; const yearMatch = sheetName.match(/\d{2,4}/); let year = yearMatch ? (parseInt(yearMatch[0]) < 100 ? 2500 + parseInt(yearMatch[0]) : parseInt(yearMatch[0])) : 2569; return new Date(year - 543, mIdx, 1); } function getAllData(sheetName) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(sheetName); if (!sheet || sheet.getLastRow() < 2) return []; const monthStart = getMonthStartDate(sheetName); const data = sheet.getRange(2, 1, sheet.getLastRow()-1, 33).getDisplayValues(); return data.map((row, index) => { let isDC = (row[19] || row[21] || row[24] || row[25] || row[26] || row[27] || row[29]); let outDateStr = row[19] || row[24] || row[25] || row[26] || row[27] || row[29]; let los = parseInt(row[22]) || 0; if (sheetName.includes("ยอดยกมาจาก") && monthStart && outDateStr) { let p = outDateStr.split('/'); let dOut = new Date(parseInt(p[2])-543, parseInt(p[1])-1, parseInt(p[0])); los = Math.floor((dOut - monthStart) / (1000 * 60 * 60 * 24)) + 1; if (los < 0) los = 0; } return { rowId: index + 2, seq: row[0], patientName: row[10], hn: row[12], ward: row[14], inDate: row[1] || row[4] || row[7], status: isDC ? "จำหน่ายแล้ว" : "ยังรักษาอยู่", displayLOS: los, rawData: row }; }); } function getMonthlyReportData(currentSheetName) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const months = ["มกราคม", "กุมภาพันธ์", "มีนาคม", "เมษายน", "พฤษภาคม", "มิถุนายน", "กรกฎาคม", "สิงหาคม", "กันยายน", "ตุลาคม", "พฤศจิกายน", "ธันวาคม"]; let results = { current: getAllData(currentSheetName), carried: [] }; // หาชีท "ยอดยกมา" ของเดือนนี้ let mIdx = -1; months.forEach((m, i) => { if (currentSheetName.includes(m)) mIdx = i; }); if (mIdx !== -1) { const carrySheetName = "ยอดยกมาจาก " + months[mIdx === 0 ? 11 : mIdx - 1]; return "อัปเดตเรียบร้อย"; } else { sheet.appendRow(rowData); return "บันทึกเรียบร้อย"; } } function closeMonthAction(payload) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const nextMonthName = payload.nextMonthName; const carryName = "ยอดยกมาจาก " + payload.currentSheetName; const currentSheet = ss.getSheetByName(payload.currentSheetName); const allData = currentSheet.getRange(2, 1, currentSheet.getLastRow()-1, 33).getDisplayValues(); const activePatients = allData.filter(r => { let isDC = (r[19] || r[21] || r[24] || r[25] || r[26] || r[27] || r[29]); return !isDC && r[10] !== ""; }); const carrySheet = currentSheet.copyTo(ss).setName(carryName); if (carrySheet.getLastRow() > 1) carrySheet.getRange(2, 1, carrySheet.getLastRow()-1, 33).clearContent(); if (activePatients.length > 0) carrySheet.getRange(2, 1, activePatients.length, 33).setValues(activePatients); const newSheet = currentSheet.copyTo(ss).setName(nextMonthName); if (newSheet.getLastRow() > 1) newSheet.getRange(2, 1, newSheet.getLastRow()-1, 33).clearContent(); return "ปิดยอดและสร้างชีทใหม่ 2 ชุดสำเร็จ"; } function createPatientPDF(data) { const html = `
